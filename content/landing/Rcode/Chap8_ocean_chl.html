    knotgrid <-spT.grid.coords(Longitude=c(maxlon-0.1, minlon+0.1),

    temp2 <- 0
    for(i in 1:length(cont)){
      temp <- cont[[i]]
      located <- point.in.polygon(knotgrid[,1],knotgrid[,2],temp$x,temp$y)
      located <- located + temp2
      temp2 <- located
    }
    ind <- located!=0
    knotgrid <- knotgrid[ind,]
